/**
 * Creates instance of Dapps API. Use *modules.api.dapps* to get existing object.
 *
 * @param cb - Callback.
 * @param _library - Object that contains helpers.
 * @constructor
 */
function Dapps(cb, _library) {
	self = this;
	library = _library;
	cb(null, self);
}
